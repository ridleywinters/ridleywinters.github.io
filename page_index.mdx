## Page Index

<table>
    <thead>
        <tr>
            <th style={{ textAlign: 'left' }}>Page</th>
            <th style={{ textAlign: 'left' }}>Path</th>
            <th style={{ textAlign: 'left' }}>Status</th>
        </tr>
    </thead>
    <tbody>
    {_.chain(database.pages)
        .sortBy((page) => `${page.path.join('/')}/${page.title}`.toLowerCase())
        .map((page) => ( 
            <tr key={page.id}>
                <td style={{ paddingRight : '2rem' }}>
                    <a href={page.href}>{page.title}</a>
                </td>
                <td style={{ paddingRight: '2rem', color: '#777' }}>
                    {page.path.join(' / ')}
                </td>
                <td style={{ paddingRight: '1rem', color: '#777' }}>
                    {page.properties.status}
                </td>            
            </tr>
        ))
        .value()}
    </tbody>
</table>
